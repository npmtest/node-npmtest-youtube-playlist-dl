function downloadVideo(id, title, callback) {
 
    var fileName = title + '.mp4';
    var video = ytdl("https://www.youtube.com/watch?v=" + id, ['--max-quality=18']);
    var output = path.join(__dirname + "/files/mp4s", fileName);
    var mp3Check = path.join(__dirname + "/files/mp3s", title + ".mp3");
 
    if (fs.existsSync(mp3Check)) {
        process.stdout.write('---Complete \n');
        callback();
    }
    else {
        var size = 0;
        video.on('error', function(err) {
            console.log("Error 2:  " + err);
            callback();
        });
 
        video.on('info', function(info) {
            size = info.size;
            video.pipe(fs.createWriteStream(output));
        });
 
        var pos = 0;
        video.on('data', function(data) {
            pos += data.length;
            if (size) {
                var percent = (pos / size * 100).toFixed(2);
                if (percent == 100) {
                    return 0;
                }
                else {
                    process.stdout.cursorTo(0);

                    process.stdout.write(percent + '%');

            }
        });
 
        video.on('end', function() {
            process.stdout.cursorTo(0);
